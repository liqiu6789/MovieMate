import requests

# 定义 API 地址
url = "http://localhost:11434/api/generate"

# 定义请求参数
payload = {
    "model": "deepseek",  # 替换为您实际使用的模型名称
    "prompt": "Once upon a time, there was a",  # 输入提示
    "max_length": 100,  # 生成的最大 token 数量
    "temperature": 0.7,  # 控制输出的随机性（值越低越确定）
}

# 发送 POST 请求
response = requests.post(url, json=payload)

# 解析响应
if response.status_code == 200:
    data = response.json()
    generated_text = data.get("response", "")
    print("Generated Text:")
    print(generated_text)
else:
    print(f"Error: {response.status_code}")
    print(response.text)
	
	
	
	
	
	import requests

# 定义 API 地址
url = "http://localhost:11434/api/models"

# 发送 GET 请求
response = requests.get(url)

# 解析响应
if response.status_code == 200:
    models = response.json()
    print("Available Models:")
    for model in models:
        print(f"Name: {model['name']}, Digest: {model['digest']}")
else:
    print(f"Error: {response.status_code}")
    print(response.text)
